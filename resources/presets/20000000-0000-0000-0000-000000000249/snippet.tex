        --++(0,\csize,0) coordinate (a)
        --++(\csize,0,0) coordinate (b)

        --++(\csize,0,0) coordinate (e) -- (b) -- cycle

        --++(0,-\csize,0) coordinate (f) edge (g) -- cycle;
